通知系统演示

点击下方按钮模拟不同类型的通知

集成示例代码

// 1. 引入通知系统
<script src="js/notifications.js"></script>
<link rel="stylesheet" href="css/notifications.css">

// 2. 初始化（在页面加载时）
const userAddress = '0x...';
const notifications = initNotificationSystem(userAddress);

// 3. 渲染铃铛图标
notifications.renderBell('notification-bell-container');

// 4. 添加通知
notifications.addNotification(
  NOTIFICATION_TYPES.REVENUE,
  '收益到账',
  '夏日微风 获得 0.1 MEER 使用费',
  { tokenId: 123, amount: '0.1' }
);

// 5. 请求桌面通知权限
notifications.requestNotificationPermission();
